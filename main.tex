\section{Two Nambu-Goto Strings}

Following the effective-string discussion of Aharony and Komargodski,
\href{https://arxiv.org/abs/1302.6257}{arXiv:1302.6257}, the leading
weight-zero term for a long string is the Nambu-Goto action, namely the
string tension times the induced worldsheet area.  For two independent
strings, let
\begin{equation}
  X_{(s)}^{\mu} = X_{(s)}^{\mu}(\sigma_s^0,\sigma_s^1),
  \qquad s=1,2,
\end{equation}
be the two spacetime embeddings.  The induced metric on the worldsheet of
string $s$ is
\begin{equation}
  h^{(s)}_{ab}
  = \partial_a X_{(s)}^{\mu}\,\partial_b X^{(s)}_{\mu},
  \qquad a,b=0,1 .
\end{equation}
The Nambu-Goto action for two decoupled strings is therefore the sum of the
two induced areas,
\begin{equation}
  S_{\mathrm{NG}}^{(1+2)}
  =
  -\sum_{s=1}^{2} T_s \int d^2\sigma_s\,
  \sqrt{-\det h^{(s)}_{ab}} .
  \label{eq:two-string-ng-covariant}
\end{equation}
Here $T_s$ is the tension of string $s$; the equal-tension case follows by
setting $T_1=T_2=T$.

In static gauge, with $X_{(s)}^0=\sigma_s^0\equiv \tau_s$ and
$X_{(s)}^1=\sigma_s^1\equiv \sigma_s$, the transverse fluctuations are
$X_{(s)}^i(\tau_s,\sigma_s)$ for $i=2,\ldots,D-1$.  Using the mostly-plus
worldsheet metric $\eta_{ab}=\operatorname{diag}(-1,+1)$, the same action
may be written as
\begin{equation}
  S_{\mathrm{NG}}^{(1+2)}
  =
  -\sum_{s=1}^{2} T_s \int d^2\sigma_s\,
  \sqrt{-\det\!\left(
    \eta_{ab}
    + \partial_a X_{(s)}^i\,\partial_b X_{(s)}^i
  \right)} .
  \label{eq:two-string-ng-static-gauge}
\end{equation}
Equivalently, the static-gauge Lagrangian density for string $s$ is
\begin{align}
  \mathcal{L}_{s}
  =
  -T_s
  \sqrt{
    \left(1-\dot{X}_{(s)}^i\dot{X}_{(s)}^i\right)
    \left(1+X_{(s)}^{j\prime}X_{(s)}^{j\prime}\right)
    +
    \left(\dot{X}_{(s)}^i X_{(s)}^{i\prime}\right)^2
  } .
  \label{eq:single-string-ng-density}
\end{align}
Dots and primes denote derivatives with respect to $\tau_s$ and $\sigma_s$,
respectively, and repeated transverse indices are summed.  If the two
worldsheets are described using common coordinates, the total Lagrangian
density is simply
\begin{equation}
  \mathcal{L}_{\mathrm{tot}} = \mathcal{L}_1 + \mathcal{L}_2 .
\end{equation}
No interaction term has been included; equations
\eqref{eq:two-string-ng-covariant}--\eqref{eq:single-string-ng-density}
represent two independent Nambu-Goto strings.

\subsection{Equal tensions and four-derivative expansion}

Now set $T_1=T_2=T$ and assume the two strings are described using the same
static-gauge worldsheet coordinates.  To avoid confusing the string label with
the sum field below, let the individual strings be labelled by $A=1,2$ in this
subsection.  Expanding the determinant form of the Nambu-Goto Lagrangian gives
for each string
\begin{align}
  \mathcal{L}_{A}
  =
  -T
  -\frac{T}{2}\,\partial_a X_A^i\,\partial^a X_A^i
  -\frac{T}{8}
    \left(\partial_a X_A^i\,\partial^a X_A^i\right)^2
  +\frac{T}{4}
    \left(\partial_a X_A^i\,\partial_b X_A^i\right)
    \left(\partial^a X_A^j\,\partial^b X_A^j\right)
  +\mathcal{O}(\partial^6) .
  \label{eq:single-string-ng-four-derivative}
\end{align}
Thus, through four-derivative order, the two-string Lagrangian density is
\begin{align}
  \mathcal{L}_{\mathrm{tot}}
  =
  -2T
  &-\frac{T}{2}\sum_{A=1}^{2}
    \partial_a X_A^i\,\partial^a X_A^i
  -\frac{T}{8}\sum_{A=1}^{2}
    \left(\partial_a X_A^i\,\partial^a X_A^i\right)^2
  \notag\\
  &+\frac{T}{4}\sum_{A=1}^{2}
    \left(\partial_a X_A^i\,\partial_b X_A^i\right)
    \left(\partial^a X_A^j\,\partial^b X_A^j\right)
  +\mathcal{O}(\partial^6) .
  \label{eq:two-string-ng-four-derivative}
\end{align}

Introduce sum and relative variables
\begin{equation}
  X_{\mathrm{s}}^i = X_1^i+X_2^i,
  \qquad
  X_{\mathrm{r}}^i = X_1^i-X_2^i,
  \qquad
  X_1^i=\frac{X_{\mathrm{s}}^i+X_{\mathrm{r}}^i}{2},
  \qquad
  X_2^i=\frac{X_{\mathrm{s}}^i-X_{\mathrm{r}}^i}{2}.
  \label{eq:sum-relative-fields}
\end{equation}
For compactness define
\begin{align}
  P_Y &\equiv \partial_a Y^i\,\partial^a Y^i,
  &
  P_{\mathrm{sr}} &\equiv
    \partial_a X_{\mathrm{s}}^i\,\partial^a X_{\mathrm{r}}^i,
  \notag\\
  G^Y_{ab} &\equiv \partial_a Y^i\,\partial_b Y^i,
  &
  G^{\mathrm{sr}}_{ab} &\equiv
    \partial_a X_{\mathrm{s}}^i\,\partial_b X_{\mathrm{r}}^i
    +\partial_a X_{\mathrm{r}}^i\,\partial_b X_{\mathrm{s}}^i ,
\end{align}
with $Y=X_{\mathrm{s}},X_{\mathrm{r}}$ and indices raised by
$\eta^{ab}$.  In terms of the sum and relative fields,
\begin{align}
  \mathcal{L}_{\mathrm{tot}}
  =
  -2T
  &-\frac{T}{4}\left(P_{X_{\mathrm{s}}}+P_{X_{\mathrm{r}}}\right)
  -\frac{T}{64}
    \left(P_{X_{\mathrm{s}}}+P_{X_{\mathrm{r}}}\right)^2
  -\frac{T}{16}P_{\mathrm{sr}}^2
  \notag\\
  &+\frac{T}{32}
    \left(G^{X_{\mathrm{s}}}_{ab}+G^{X_{\mathrm{r}}}_{ab}\right)
    \left(G_{X_{\mathrm{s}}}^{ab}+G_{X_{\mathrm{r}}}^{ab}\right)
  +\frac{T}{32}
    G^{\mathrm{sr}}_{ab}G_{\mathrm{sr}}^{ab}
  +\mathcal{O}(\partial^6) .
  \label{eq:two-string-ng-sum-relative}
\end{align}
The factors of $1/4$, $1/16$, and $1/64$ arise because the variables in
\eqref{eq:sum-relative-fields} are not normalized by $1/\sqrt{2}$.

\subsection{Shift currents}

The Lagrangian in \eqref{eq:two-string-ng-sum-relative} depends only on
derivatives of $X_{\mathrm{s}}^i$ and $X_{\mathrm{r}}^i$.  It therefore has
independent constant-shift symmetries
\begin{equation}
  X_{\mathrm{s}}^i \to X_{\mathrm{s}}^i+\epsilon_{\mathrm{s}}^i,
  \qquad
  X_{\mathrm{r}}^i \to X_{\mathrm{r}}^i+\epsilon_{\mathrm{r}}^i .
\end{equation}
The associated Noether currents are obtained by differentiating
\eqref{eq:two-string-ng-sum-relative} with respect to the corresponding field
derivatives,
\begin{equation}
  J_{\mathrm{s}}^{a i}
  \equiv
  \frac{\partial\mathcal{L}_{\mathrm{tot}}}
       {\partial(\partial_a X_{\mathrm{s}}^i)},
  \qquad
  J_{\mathrm{r}}^{a i}
  \equiv
  \frac{\partial\mathcal{L}_{\mathrm{tot}}}
       {\partial(\partial_a X_{\mathrm{r}}^i)} .
\end{equation}
Through the same four-derivative order, these currents are
\begin{align}
  J_{\mathrm{s}}^{a i}
  =
  &-\frac{T}{2}\,\partial^a X_{\mathrm{s}}^i
  -\frac{T}{16}
    \left(P_{X_{\mathrm{s}}}+P_{X_{\mathrm{r}}}\right)
    \partial^a X_{\mathrm{s}}^i
  -\frac{T}{8}P_{\mathrm{sr}}\,
    \partial^a X_{\mathrm{r}}^i
  \notag\\
  &+\frac{T}{8}
    \left(G_{X_{\mathrm{s}}}^{ab}+G_{X_{\mathrm{r}}}^{ab}\right)
    \partial_b X_{\mathrm{s}}^i
  +\frac{T}{8}
    G_{\mathrm{sr}}^{ab}\partial_b X_{\mathrm{r}}^i
  +\mathcal{O}(\partial^5),
  \label{eq:shift-current-sum}
  \\
  J_{\mathrm{r}}^{a i}
  =
  &-\frac{T}{2}\,\partial^a X_{\mathrm{r}}^i
  -\frac{T}{16}
    \left(P_{X_{\mathrm{s}}}+P_{X_{\mathrm{r}}}\right)
    \partial^a X_{\mathrm{r}}^i
  -\frac{T}{8}P_{\mathrm{sr}}\,
    \partial^a X_{\mathrm{s}}^i
  \notag\\
  &+\frac{T}{8}
    \left(G_{X_{\mathrm{s}}}^{ab}+G_{X_{\mathrm{r}}}^{ab}\right)
    \partial_b X_{\mathrm{r}}^i
  +\frac{T}{8}
    G_{\mathrm{sr}}^{ab}\partial_b X_{\mathrm{s}}^i
  +\mathcal{O}(\partial^5).
  \label{eq:shift-current-relative}
\end{align}
The equations of motion can be written as current conservation laws,
\begin{equation}
  \partial_a J_{\mathrm{s}}^{a i}=0,
  \qquad
  \partial_a J_{\mathrm{r}}^{a i}=0.
\end{equation}
